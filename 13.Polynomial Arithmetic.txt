#include <stdio.h>
#define MAX 10

int main() {
    int poly1[MAX], poly2[MAX], result[MAX];
    int deg1, deg2, i;
    
    printf("Enter degree of first polynomial: ");
    scanf("%d", &deg1);
    printf("Enter coefficients of first polynomial (highest to lowest degree):\n");
    for (i = 0; i <= deg1; i++) {
        scanf("%d", &poly1[i]);
    }

    printf("Enter degree of second polynomial: ");
    scanf("%d", &deg2);
    printf("Enter coefficients of second polynomial (highest to lowest degree):\n");
    for (i = 0; i <= deg2; i++) {
        scanf("%d", &poly2[i]);
    }

    /* Polynomial Addition */
    int maxDeg = (deg1 > deg2) ? deg1 : deg2;
    int diff;

    for (i = 0; i <= maxDeg; i++) {
        result[i] = 0;
    }

    /* Add poly1 coefficients */
    diff = maxDeg - deg1;
    for (i = 0; i <= deg1; i++) {
        result[i + diff] += poly1[i];
    }

    /* Add poly2 coefficients */
    diff = maxDeg - deg2;
    for (i = 0; i <= deg2; i++) {
        result[i + diff] += poly2[i];
    }

    printf("Sum of polynomials: ");
    for (i = 0; i <= maxDeg; i++) {
        printf("%d", result[i]);
        if (i != maxDeg) {
            printf("x^%d + ", maxDeg - i);
        }
    }
    printf("\n");

    /* Polynomial Subtraction (poly1 - poly2) */
    for (i = 0; i <= maxDeg; i++) {
        result[i] = 0;
    }

    diff = maxDeg - deg1;
    for (i = 0; i <= deg1; i++) {
        result[i + diff] += poly1[i];
    }

    diff = maxDeg - deg2;
    for (i = 0; i <= deg2; i++) {
        result[i + diff] -= poly2[i];
    }

    printf("Difference of polynomials (poly1 - poly2): ");
    for (i = 0; i <= maxDeg; i++) {
        printf("%d", result[i]);
        if (i != maxDeg) {
            printf("x^%d + ", maxDeg - i);
        }
    }
    printf("\n");

    return 0;
}
