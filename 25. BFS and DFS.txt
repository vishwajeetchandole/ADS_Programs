#include <stdio.h>
#include <stdlib.h>

#define MAX 100

int adj[MAX][MAX], visited[MAX], queue[MAX];
int front = -1, rear = -1;

void enqueue(int v) {
    if (rear == MAX - 1) return;
    if (front == -1) front = 0;
    rear++;
    queue[rear] = v;
}

int dequeue() {
    int v;
    if (front == -1 || front > rear) return -1;
    v = queue[front];
    front++;
    return v;
}

void BFS(int start, int n) {
    int i, v;
    for (i = 0; i < n; i++) visited[i] = 0;
    enqueue(start);
    visited[start] = 1;
    while (front <= rear) {
        v = dequeue();
        printf("%d ", v);
        for (i = 0; i < n; i++) {
            if (adj[v][i] == 1 && visited[i] == 0) {
                enqueue(i);
                visited[i] = 1;
            }
        }
    }
}

void DFS(int start, int n) {
    int i;
    printf("%d ", start);
    visited[start] = 1;
    for (i = 0; i < n; i++) {
        if (adj[start][i] == 1 && visited[i] == 0)
            DFS(i, n);
    }
}

int main() {
    int n, e, i, j, src, dest, start;
    printf("Enter number of vertices: ");
    scanf("%d", &n);
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            adj[i][j] = 0;
        }
    }
    printf("Enter number of edges: ");
    scanf("%d", &e);
    for (i = 0; i < e; i++) {
        printf("Enter edge (source destination): ");
        scanf("%d %d", &src, &dest);
        adj[src][dest] = 1;
        adj[dest][src] = 1;
    }
    printf("Enter starting vertex: ");
    scanf("%d", &start);
    printf("BFS Traversal: ");
    BFS(start, n);
    for (i = 0; i < n; i++) visited[i] = 0;
    printf("\nDFS Traversal: ");
    DFS(start, n);
    printf("\n");
    return 0;
}
